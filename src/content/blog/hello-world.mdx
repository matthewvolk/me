### With Filename

Use `// @filename:` to display a title:

```tsx
// @filename: components/Button.tsx
export function Button({ children }: { children: React.ReactNode }) {
  return (
    <button className="rounded bg-primary px-4 py-2 text-primary-foreground">
      {children}
    </button>
  );
}
```

### Line Highlighting

Use `// [!code highlight]` to highlight specific lines:

```ts
// @filename: lib/api.ts
async function fetchData(url: string) {
  const response = await fetch(url); // [!code highlight]
  if (!response.ok) {
    throw new Error("Failed to fetch"); // [!code highlight]
  }
  return response.json();
}
```

### Diff Style

Use `// [!code ++]` and `// [!code --]` for diff highlighting:

```tsx
// @filename: app/page.tsx
export default async function Page() {
  const data = await fetchData("/api/posts");
  const items = data.items; // [!code --]
  const { items, total } = data; // [!code ++]

  return (
    <ul>
      {items.map((post) => (
        <li key={post.id}>{post.title}</li>
      ))}
    </ul>
  );
}
```
